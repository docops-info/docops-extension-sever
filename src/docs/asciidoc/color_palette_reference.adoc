// Available palette types:
- CORPORATE          // Professional business colors

- TABLEAU            // Industry-standard (default light)
- MATERIAL           // Google Material Design
- PASTEL             // Soft, muted tones
- MONOCHROMATIC_BLUE // Blue gradient variations
- SUNSET_VIBRANT     // Warm, energetic colors
- FOREST_DEPTHS      // Natural green palette
- OCEAN_BREEZE       // Coastal blue-green
- COSMIC_NEBULA      // Deep space purples
- URBAN_NIGHT        // Sophisticated dark (default dark)
- TROPICAL_PARADISE  // Bold tropical colors
